**test_pdf_extractor.py示例**：
```python
import pytest
from modules.pdf_extractor import pdf_extractor

def test_extract_text(test_pdf_path):
    """测试PDF文本提取功能"""
    result = pdf_extractor.extract_text(test_pdf_path)
    
    # 验证结果结构
    assert 'total_pages' in result
    assert 'text_content' in result
    assert 'tables' in result
    
    # 验证页面数量大于0
    assert result['total_pages'] > 0
    
    # 验证文本内容存在
    assert len(result['text_content']) == result['total_pages']
    
    # 验证至少有一个文本块
    assert any(len(page['text_blocks']) > 0 for page in result['text_content'])
```

#### 2.2 重写 `extract_tables` 方法

使用 camelot-py 的 API 替换 pdfplumber 的实现，保持相同的方法签名和返回值格式：

```python
def extract_tables(self, pdf_path, pages=None):
    """提取PDF中的表格内容，可以指定页面

    Args:
        pdf_path (str): PDF文件路径
        pages (list[int] | None): 指定要提取的页码列表（从1开始），None表示提取所有页面
        
    Returns:
        tuple[list[PdfTable], dict]: 包含提取的表格列表和按页码组织的表格边界框字典
            - list[PdfTable]: 提取的表格列表
            - dict: 按页码组织的表格边界框字典，键为页码，值为边界框列表
    """
    if not os.path.exists(pdf_path):
        raise FileNotFoundError(f"PDF文件不存在: {pdf_path}")
    
    pdf_tables = []
    page_tables = {}
    
    try:
        logger.info(f"开始提取PDF表格: {pdf_path}")
        
        # 构建页码参数
        pages_param = None
        if pages is not None:
            pages_param = ','.join(map(str, pages))
        
        # 使用camelot提取表格
        # 尝试使用lattice模式（基于线条），如果失败则使用stream模式（基于空白）
        try:
            tables = camelot.read_pdf(pdf_path, pages=pages_param, flavor='lattice')
            logger.info(f"使用lattice模式提取表格: 成功提取{len(tables)}个表格")
        except Exception as e:
            logger.warning(f"lattice模式提取失败: {e}，尝试使用stream模式")
            tables = camelot.read_pdf(pdf_path, pages=pages_param, flavor='stream')
            logger.info(f"使用stream模式提取表格: 成功提取{len(tables)}个表格")
        
        # 处理提取的表格
        for table_idx, table in enumerate(tables):
            page_num = table.page
            
            # 获取表格边界框
            try:
                if hasattr(table, 'parsing_report') and 'bbox' in table.parsing_report:
                    bbox = table.parsing_report['bbox']
                elif hasattr(table, '_bbox'):
                    bbox = table._bbox
                else:
                    bbox = None
                logger.info(f"表格{table_idx}边界框: {bbox}")
            except Exception as e:
                logger.warning(f"获取表格边界框失败: {e}")
                bbox = None
            
            # 获取表格内容
            data = table.data
            logger.info(f"表格{table_idx}内容: {data}")
            
            # 构建单元格信息
            cell_info_list = []
            for row_idx, row in enumerate(data):
                for col_idx, text in enumerate(row):
                    # 估算单元格边界框
                    if bbox:
                        table_x0, table_y0, table_x1, table_y1 = bbox
                        num_rows = len(data)
                        num_cols = len(row)
                        
                        # 计算单元格大小和位置
                        cell_height = (table_y1 - table_y0) / num_rows
                        cell_width = (table_x1 - table_x0) / num_cols
                        
                        x0 = table_x0 + col_idx * cell_width
                        y0 = table_y0 + row_idx * cell_height
                        x1 = x0 + cell_width
                        y1 = y0 + cell_height
                        cell_bbox = (x0, y0, x1, y1)
                    else:
                        cell_bbox = (0, 0, 100, 30)  # 默认值
                    
                    cell_info = {
                        'text': text or '',
                        'bbox': cell_bbox,
                        'top': row_idx,
                        'left': col_idx,
                        'x0': cell_bbox[0],
                        'y0': cell_bbox[1],
                        'x1': cell_bbox[2],
                        'y1': cell_bbox[3],
                        'width': cell_bbox[2] - cell_bbox[0],
                        'height': cell_bbox[3] - cell_bbox[1]
                    }
                    cell_info_list.append(cell_info)
                    logger.debug(f"创建单元格信息: 行={row_idx}, 列={col_idx}, 文本='{text}', 边界框={cell_bbox}")
            
            # 构建单元格二维列表
            if cell_info_list:
                max_row = max(cell['top'] for cell in cell_info_list)
                max_col = max(cell['left'] for cell in cell_info_list)
                logger.info(f"表格结构: {max_row + 1}行 {max_col + 1}列")
            else:
                max_row = 0
                max_col = 0
                logger.info("表格无内容")
            
            # 创建空的单元格二维列表
            cell_matrix = [[None for _ in range(max_col + 1)] for _ in range(max_row + 1)]
            
            # 填充单元格信息
            for cell_info in cell_info_list:
                row_idx = cell_info['top']
                col_idx = cell_info['left']
                text = cell_info['text'] if cell_info['text'] else ''
                cell_bbox = cell_info['bbox'] if 'bbox' in cell_info else (cell_info['x0'], cell_info['y0'], cell_info['x1'], cell_info['y1'])
                
                # 创建PdfCell对象
                pdf_cell = PdfCell(
                    text=text,
                    bbox=cell_bbox,
                    row_idx=row_idx,
                    col_idx=col_idx
                )
                
                cell_matrix[row_idx][col_idx] = pdf_cell
                logger.debug(f"填充单元格: 行={row_idx}, 列={col_idx}, 文本='{text}', 边界框={cell_bbox}")
            
            # 过滤掉空表格
            has_content = any(cell and cell.text.strip() for row in cell_matrix for cell in row)
            if has_content:
                # 计算行高和列宽
                row_heights = []
                for row_idx in range(len(cell_matrix)):
                    max_height = 0
                    for cell in cell_matrix[row_idx]:
                        if cell:
                            max_height = max(max_height, cell.height)
                    if max_height > 0:
                        row_heights.append(max_height)
                    else:
                        row_heights.append(40)  # 默认行高
                
                col_widths = []
                if cell_matrix:
                    num_cols = len(cell_matrix[0])
                    for col_idx in range(num_cols):
                        max_width = 0
                        for row in cell_matrix:
                            if col_idx < len(row) and row[col_idx]:
                                max_width = max(max_width, row[col_idx].width)
                        if max_width > 0:
                            col_widths.append(max_width)
                        else:
                            if bbox:
                                table_width = bbox[2] - bbox[0]
                                col_widths.append(table_width / num_cols)
                            else:
                                col_widths.append(100)  # 默认列宽
                
                logger.info(f"计算的行高: {row_heights}")
                logger.info(f"计算的列宽: {col_widths}")
                
                # 创建PdfTable对象并添加到pdf_tables列表
                pdf_table = PdfTable(
                    page_num=page_num,
                    table_idx=table_idx,
                    cells=cell_matrix,
                    bbox=bbox,
                    row_heights=row_heights,
                    col_widths=col_widths
                )
                pdf_tables.append(pdf_table)
                logger.info(f"表格: 第{page_num}页-表格{table_idx}, 包含{len(cell_matrix)}行{len(cell_matrix[0]) if cell_matrix else 0}列, 边界框: {bbox}")
                
                # 按页码组织表格数据
                if page_num not in page_tables:
                    page_tables[page_num] = []
                if bbox:
                    page_tables[page_num].append(bbox)
        
        logger.info(f"表格提取完成: 总表格={len(pdf_tables)}")
        return pdf_tables, page_tables
        
    except Exception as e:
        logger.error(f"提取PDF表格时出错: {str(e)}", exc_info=True)
        raise Exception(f"提取PDF表格时出错: {str(e)}")

### 3. 表格结构和单元格大小设计

#### 3.1 表格结构

使用 camelot-py 提取的表格结构如下：
- 表格对象包含 `data` 属性，是一个二维列表，表示表格的行和列
- 表格对象包含 `page` 属性，表示表格所在的页码
- 表格对象可能包含 `parsing_report` 属性，其中包含表格的边界框信息

#### 3.2 单元格大小计算

基于 camelot-py 提供的表格边界框和行列数，计算每个单元格的大小：

1. **获取表格边界框**：从表格对象的 `parsing_report` 或其他属性中获取整个表格的边界框
2. **计算表格尺寸**：边界框的宽度和高度
3. **计算单元格大小**：
   - 单元格宽度 = 表格宽度 / 列数
   - 单元格高度 = 表格高度 / 行数
4. **计算单元格位置**：
   - 单元格 x0 = 表格 x0 + 列索引 * 单元格宽度
   - 单元格 y0 = 表格 y0 + 行索引 * 单元格高度
   - 单元格 x1 = 单元格 x0 + 单元格宽度
   - 单元格 y1 = 单元格 y0 + 单元格高度

#### 3.3 优势

- **更准确的表格边界框**：camelot-py 提供的表格边界框通常比 pdfplumber 更准确
- **更好的表格识别**：camelot-py 在处理复杂表格时表现更好
- **更灵活的提取模式**：支持 lattice 和 stream 两种模式，适应不同类型的表格
